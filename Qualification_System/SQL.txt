CREATE TABLE STUDENT ( 
SID INT(7) NOT NULL,  
SNAME VARCHAR(20) NOT NULL, 
SKNAME VARCHAR(20) NOT NULL , 
SEMAIL VARCHAR(40) NOT NULL, 
DEPARTMENT_ID INT(2) NOT NULL, 
COURSE_ID INT(2) NOT NULL, 
SCHOOL_YEAR INT(1) NOT NULL,
CLASS INT(1) NOT NULL, 
QUESTION INT(2) NOT NULL,
ANSWER VARCHAR(30) NOT NULL,
PASSWORD VARCHAR(40)  NOT NULL, 
INDEX(SID), 
PRIMARY KEY(SID),
FOREIGN KEY (DEPARTMENT_ID)
REFERENCES DEPARTMENT(DEPARTMENT_ID)
on update cascade,
FOREIGN KEY (DEPARTMENT_ID,COURSE_ID)
REFERENCES COURSE(DEPARTMENT_ID,COURSE_ID)
on update cascade
)ENGINE=InnoDB;

CREATE TABLE QUALIFICATION( 
QID INT(9) NOT NULL AUTO_INCREMENT, 
CID INT(6) NOT NULL, 
CNAME VARCHAR(20) NOT NULL, 
DIFFICULTY VARCHAR(1) NOT NULL,
SID INT(7) NOT NULL,  
DEPARTMENT_ID INT(2), 
COURSE_ID INT(2), 
SCHOOL_YEAR INT(1),
CLASS INT(1), 
REGISTRATION_DATE DATE NOT NULL, 
UPDATE_DATE DATETIME NOT NULL,
EXAM_DATE DATE NOT NULL, 
RESULT VARCHAR(3) NOT NULL, 
PRIMARY KEY (QID),
FOREIGN KEY (CID)
REFERENCES CREDENTIAL_INFORMATION(CID)
ON UPDATE CASCADE,
FOREIGN KEY (SID)
REFERENCES student(SID)
ON UPDATE CASCADE,
FOREIGN KEY (DEPARTMENT_ID)
REFERENCES DEPARTMENT(DEPARTMENT_ID)
ON UPDATE CASCADE ON DELETE SET NULL,
FOREIGN KEY (DEPARTMENT_ID,COURSE_ID)
REFERENCES COURSE(DEPARTMENT_ID,COURSE_ID)
ON UPDATE CASCADE ON DELETE SET NULL
)ENGINE=InnoDB; 
受験情報

CREATE TABLE CREDENTIAL_INFORMATION( 
CID INT(6) NOT NULL, 
CNAME VARCHAR(20) NOT NULL, 
CLASSIFICATION_ID INT(4), 
DIFFICULTY VARCHAR(1) NOT NULL, 
SUPERIOR1 INT(6),
SUPERIOR2 INT(6),
SUPERIOR3 INT(6),
SUPERIOR4 INT(6),
LOW_ORDER1 INT(6), 
LOW_ORDER2 INT(6), 
LOW_ORDER3 INT(6), 
LOW_ORDER4 INT(6), 
PRIMARY KEY (CID) ,
FOREIGN KEY (CLASSIFICATION_ID )
REFERENCES CLASSIFICATION(CLASSIFICATION_ID)
ON DELETE SET NULL
on update cascade
)ENGINE=InnoDB;
